% 2) Problématique et besoin
\begin{frame}{Problématique et besoin}
  \begin{itemize}
    \item Paiement lent et peu traçable (buvette/restaurant).
    \item Besoin d’un système simple: scan produits + paiement RFID.
    \item Besoin de \textbf{2 modes stricts} : RESTO (ticket fixe) et BUVETTE (panier).
  \end{itemize}
\end{frame}

% 3) Utilisateurs cibles
\begin{frame}{Utilisateurs cibles}
  \begin{itemize}
    \item \textbf{Étudiant} : solde, carte RFID, historique, paiements.
    \item \textbf{Admin} : étudiants, produits, cartes RFID, recharges.
    \item \textbf{Terminal/Opérateur} : validation RFID et monitoring.
  \end{itemize}
\end{frame}

% 4) Objectifs
\begin{frame}{Objectifs}
  \begin{itemize}
    \item Paiement rapide et fiable.
    \item Carte étudiante \textbf{smart} : identification + accès au solde + paiement.
    \item Traçabilité (référence + solde avant/après).
    \item Sécurité minimale: sessions + CSRF sur actions sensibles.
    \item Intégration ESP32 + RFID + scan webcam.
  \end{itemize}
\end{frame}

% 5) Étude / scénarios
\begin{frame}{Scénarios principaux}
  \textbf{BUVETTE}
  \begin{itemize}
    \item Scan code-barres \(\rightarrow\) panier \(\rightarrow\) commande \textit{PENDING}
    \item ESP32 (mode \textbf{BUVETTE} synchronisé depuis le site) réclame la commande \(\rightarrow\) débite \(\rightarrow\) statut PAYE/ECHEC
  \end{itemize}
  \vspace{6pt}
  \textbf{RESTO}
  \begin{itemize}
    \item ESP32 (mode \textbf{RESTO} synchronisé depuis le site) lit UID \(\rightarrow\) débit ticket fixe \(\rightarrow\) transfert enregistré
  \end{itemize}
\end{frame}

% 9) Choix techniques
\begin{frame}{Choix techniques}
  \begin{itemize}
    \item Backend: PHP (PDO), sessions, CSRF.
    \item Base de données: MySQL (XAMPP).
    \item Barcode: BarcodeDetector + ZXing + Quagga fallback (optimisé).

% 11) Réalisation
\begin{frame}{Réalisation (fonctionnalités)}
  \begin{itemize}
    \item Admin: ajout étudiant, ajout produit (code-barres), assignation carte, recharge.
    \item Étudiant: solde, carte, historique, paiement resto, paiement buvette (scan).
    \item Terminal: RESTO (ticket fixe) + BUVETTE (commande + RFID) + Serial Monitor web.

% 12) Résultats
\begin{frame}{Résultats obtenus}
  \begin{itemize}
    \item Paiement RESTO opérationnel (transferts enregistrés).
    \item Paiement BUVETTE opérationnel (commande PENDING → PAYE/ECHEC).
    \item Interfaces unifiées + monitoring série.
  \end{itemize}
\end{frame}

% 13) Tests et validation
\begin{frame}{Tests et validation}
  \begin{itemize}
    \item Scan: produit connu/inconnu, stabilité et performance.
    \item Paiement: solde insuffisant, carte inactive, erreurs réseau.
    \item Concurrence: transaction + verrouillage solde (FOR UPDATE).
  \end{itemize}
\end{frame}

% 14) Difficultés
\begin{frame}{Difficultés rencontrées}
  \begin{itemize}
    \item Caméra bloquée en HTTP non sécurisé (solution: localhost/HTTPS).
    \item Timeouts ESP32 (solution: IP correcte + timeouts + debug logs).
    \item Optimisation scan (formats limités, zone centrale, cooldown).
    \item Câblage LED inversé possible (active LOW) \(\rightarrow\) correction par recâblage ou logique firmware.
  \end{itemize}
\end{frame}

\begin{frame}{Conclusion et perspectives}

    \item Perspectives: durcissement sécurité, ergonomie mobile, reporting admin, meilleure gestion HTTPS.
  \end{itemize}
\end{frame}
